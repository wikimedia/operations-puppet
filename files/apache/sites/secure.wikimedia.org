<VirtualHost *:443>
	ServerName secure.wikimedia.org
	ServerAdmin noc@wikimedia.org

	SSLEngine on
	SSLCertificateFile /etc/ssl/certs/star.wikimedia.org.crt
	SSLCertificateKeyFile /etc/ssl/private/star.wikimedia.org.key

	ErrorLog /var/log/apache2/error.log

	# Possible values include: debug, info, notice, warn, error, crit,
	# alert, emerg.
	LogLevel warn

	CustomLog /var/log/apache2/access.log combined

	# Proxy secure.wikimedia.org to the Apaches
	<Proxy *>
		Allow from all
	</Proxy>

	ProxyRequests Off
	ProxyPreserveHost On
	ProxyPass /otrs !
	ProxyPass /otrs-2.1.3 !
	ProxyPass /otrs-web !
	ProxyPass / http://apaches.pmtpa.wmnet/ retry=3
	SetEnv force-proxy-request-1.0 1
	SetEnv proxy-nokeepalive 1
	SetEnv proxy-initial-not-pooled
	ProxyVia On

	# Old OTRS URLs
	RewriteEngine On
	RewriteRule ^/otrs/(.*)$ https://ticket.wikimedia.org/otrs/$1 [R=301,L]
	RewriteRule ^/otrs-web/(.*)$ https://ticket.wikimedia.org/otrs-web/$1 [R=301,L]

	# old URLs, redirect them to proper SSL
	RewriteRule ^/wikipedia/(advisory|auditcom|boardgovcom|board|chair|chapcom|checkuser|collab|commons|donate|exec|grants|incubator|internal|meta|movementroles|office|otrs-wiki|outreach|quality|searchcom|spcom|species|steward|strategy|usability|wikimania\d+|wikimaniateam)/(.*)$ https://$1.wikimedia.org/$2 [R=302,L]
	RewriteRule ^/wikipedia/foundation/(.*)$ https://wikimediafoundation.org/$1 [R=302,L]
	RewriteRule ^/wikipedia/mediawiki/(.*)$ https://www.mediawiki.org/$1 [R=302,L]
	RewriteRule ^/wikipedia/sources/(.*)$ https://wikisource.org/$1 [R=302,L]
	RewriteRule ^/(wikipedia|wikinews|wikisource|wikibooks|wikiquote|wikiversity|wiktionary|wikimedia|wikidata)/([^@:/]+)/(.*)$ https://$2.$1.org/$3 [R=302,L]

</VirtualHost>

<VirtualHost *:80>
	ServerName secure.wikimedia.org

	RewriteEngine on
	RewriteRule ^/(.*)$ https://secure.wikimedia.org/$1 [R=302]
</VirtualHost>
